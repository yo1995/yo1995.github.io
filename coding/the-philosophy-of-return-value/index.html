class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int> result;
        int index1 = 0;
        int index2 = 1;
        std::vector<int>::iterator it1 = nums.begin();
        std::vector<int>::iterator it2 = nums.begin() + 1;
        while (it1 != nums.end()) {
            while(it2 != nums.end()) {
                if (*it1 + *it2 == target) {
                    result.push_back(index1); 
                    result.push_back(index2); 
                    return result;
                }
                else {
                    ++it2;
                    ++index2;
                }
            }
            ++it1;++index1;
            it2 = it1 + 1; // reset the position
            index2 = index1+1;
        }
    };
};
